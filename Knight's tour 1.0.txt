#include "pch.h"
#include <iostream>
using namespace std;
#define MAX 100 //Ngan x?p ch?a t?i da 100 ph?n t?

struct STACK {
	int a[MAX];	//M?ng các ph?n t? c?a ngan x?p
	int top;	//Ch? s? ph?n t? d?nh d?u ngan x?p
};

//1. T?o ngan x?p r?ng
void InitStack(STACK&myStack);
//2. Ki?m tra ngan x?p r?ng
int IsEmptyStack(STACK&myStack);
//3. Ki?m tra ngan x?p d?y hay ko?
int IsFullStack(STACK&myStack);
//4. Thêm 1 ph?n t? vào ngan x?p
void Push(STACK&myStack, int x);
//5. L?y thông tin ph?n t? ? d?nh ngan x?p
int Top(STACK myStack);
//6. Trích h?y ph?n t? ? d?nh ngan x?p
int Pop(STACK&myStack);

int main() {
	STACK s;
	InitStack(s);

}

//1. T?o ngan x?p r?ng
void InitStack(STACK&myStack) {
	myStack.top = 1;
}
//2. Ki?m tra ngan x?p r?ng
int IsEmptyStack(STACK&myStack) {
	if (myStack.top == -1) return 1;//Ngan x?p r?ng
	return 0;						//Ngan x?p không r?ng
}
//3. Ki?m tra ngan x?p d?y hay ko?
int IsFullStack(STACK&myStack) {
	if (myStack.top == MAX - 1) return 1;	//Ngan x?p d?y
	return 0;								//Ngan x?p chua d?y
}
//4. Thêm 1 ph?n t? vào ngan x?p
void Push(STACK&myStack, int x) {
	if (IsFullStack(myStack)) cout << "Ngan xep day";
	else {
		myStack.a[myStack.top + 1] = x;
		myStack.top++;
	}
}
//5. L?y thông tin ph?n t? ? d?nh ngan x?p
int Top(STACK myStack) {
	return myStack.a[myStack.top];
}
//6. Trích h?y ph?n t? ? d?nh ngan x?p
int Pop(STACK&myStack) {
	int t = myStack.a[myStack.top];
	myStack.top--;
	return t;
}
